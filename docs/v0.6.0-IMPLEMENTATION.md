## Feature #1: CACHING LAYER

### What It Does

Stores review results by content hash. Second request for same content returns instantly from cache (5ms, $0 cost).

### Import & Usage

```typescript
import { CacheManager, type CacheConfig } from 'cross-review-mcp';

// Initialize cache
const cacheConfig: CacheConfig = {
  enabled: true,
  ttl: 86400,      // 24 hours
  maxSize: 1000    // max entries
};

const cache = new CacheManager(cacheConfig);

// Set cache entry
cache.set('Your code...', { cost: 0.0049, time: 1250 });

// Get from cache (returns null if missing or expired)
const cached = cache.get('Your code...');
if (cached) {
  console.log('Cache hit!', cached); // 5ms, no API call
} else {
  console.log('Cache miss - call API');
}

// Check without modifying
if (cache.has('Your code...')) {
  // Will call cache.get() next
}

// View stats
console.log(cache.getStats());
// {
//   size: 1,
//   maxSize: 1000,
//   ttl: 86400,
//   entries: [...]
// }
```

### Real-World Example

```typescript
// Batch of 100 reviews (70 unique, 30 duplicates)
for (const review of reviews) {
  const hash = review.content; // simplified
  
  // Check cache first
  let result = cache.get(review.content);
  
  if (!result) {
    // Cache miss - call API
    result = await executor.execute(review);
    cache.set(review.content, result);
  }
  
  console.log(`Review: $${result.cost}, ${result.time}ms`);
}

// Results:
// - 70 API calls (not 100!)
// - Cost: 70 × $0.0049 = $0.343 (saved $0.147)
// - Time: 80 seconds (vs 130 without cache)
```

## Feature #2: RATE LIMITING

### What It Does

Token bucket algorithm prevents exceeding API quotas. If quota exhausted, transparently waits until tokens refill.

### Import & Usage

```typescript
import { RateLimiter, type RateLimitConfig } from 'cross-review-mcp';

// Initialize rate limiter
const rateLimitConfig = {
  openai: { requestsPerMinute: 100 },
  gemini: { requestsPerMinute: 50 }
};

const limiter = new RateLimiter(rateLimitConfig);

// Before calling API, check rate limit
await limiter.waitIfNeeded('openai');
// If quota available: returns immediately
// If quota full: waits ~0.6 seconds, then returns

// Make API call
const result = await executeReview(content);

// View state
const state = limiter.getState('openai');
console.log(state);
// {
//   tokensAvailable: 85,
//   requestsPerMinute: 100,
//   nextTokenRefillTime: Date
// }
```

### Real-World Example

```typescript
// Batch of 300 reviews (exceeds 1-minute quota of 100)
for (const review of reviews) {
  // Rate limiting handled transparently
  await limiter.waitIfNeeded('openai');
  
  const result = await callOpenAI(review);
  console.log(result);
}

// Without rate limiting: 429 errors after ~50 reviews
// With rate limiting:
// - Review 1-100: Immediate (uses quota)
// - Review 101-150: Wait ~0.6s each (tokens refill)
// - Review 151-200: Immediate (quota refilled)
// - ... continues safely
//
// Result: 0 errors, all 300 reviews processed ✅
```

## Feature #3: BATCH PROCESSING

### What It Does

Async job queue with 4 parallel workers. Submit 100 reviews once, get back a `batchId`, track progress, retrieve results when complete.

### Import & Usage

```typescript
import { BatchProcessor, type BatchProcessorConfig } from 'cross-review-mcp';

// Initialize batch processor
const batchConfig: BatchProcessorConfig = {
  parallelWorkers: 4,
  workerTimeoutMs: 120000,  // 2 min timeout per item
  maxBatchSize: 10000
};

const processor = new BatchProcessor(batchConfig);

// Define work function
async function reviewItem(item: string, workerIndex: number) {
  console.log(`Worker ${workerIndex} processing: ${item}`);
  const result = await executor.execute({ content: item });
  return result;
}

// Submit batch (returns immediately)
const reviews = [
  'Review this...',
  'Check this...',
  // ... 98 more
];

const batchId = await processor.submitBatch(
  reviews,
  reviewItem,
  'https://myserver.com/webhook'  // optional
);

console.log(`Batch submitted: ${batchId}`);
// Returns immediately, processing happens in background
```

### Poll for Status

```typescript
// Check progress
const status = processor.getStatus(batchId);
console.log(status);
// {
//   id: 'batch-1710....',
//   status: 'PROCESSING',
//   totalItems: 100,
//   completedItems: 45,
//   failedItems: 0,
//   createdAt: Date,
//   completedAt: null,
//   errors: []
// }

// Poll until complete
while (true) {
  const status = processor.getStatus(batchId);
  
  if (status?.status === 'COMPLETE') {
    const results = processor.getResults(batchId);
    console.log(`Completed ${results?.length} reviews`);
    break;
  }
  
  console.log(`Progress: ${status?.completedItems}/${status?.totalItems}`);
  await new Promise(r => setTimeout(r, 5000)); // Check every 5s
}
```

### Real-World Example

```typescript
// 100 reviews submitted
const batchId = await processor.submitBatch(reviews, reviewItem);
console.log(`Batch ${batchId} queued`);

// Timeline (with 4 workers):
// Time  Status

// 1250ms - First batch of 25 complete
// 2500ms - Second batch complete
// 3750ms - Third batch complete
// 5000ms - Fourth batch complete (all 100 done)
//
// Total time: ~5 seconds (vs 130 seconds sequential!)
```

## 🔗 Combining All Three

### Real Production Scenario

```typescript
import {
  ReviewExecutor,
  CacheManager,
  RateLimiter,
  BatchProcessor
} from 'cross-review-mcp';

const cache = new CacheManager({ enabled: true, ttl: 86400, maxSize: 1000 });
const limiter = new RateLimiter({
  openai: { requestsPerMinute: 100 },
  gemini: { requestsPerMinute: 50 }
});
const processor = new BatchProcessor({
  parallelWorkers: 4,
  workerTimeoutMs: 120000,
  maxBatchSize: 10000
});

// Custom review function with all three features
async function reviewWithAllFeatures(
  items: string[],
  webhookUrl?: string
): Promise<string> {
  // Submit batch
  const batchId = await processor.submitBatch(
    items,
    async (item) => {
      // Check cache first
      const cached = cache.get(item);
      if (cached) {
        return cached; // Instant, $0
      }

      // Respect rate limits
      await limiter.waitIfNeeded('openai');

      // Call API
      const result = await executor.execute({
        content: item
      });

      // Store in cache for future use
      cache.set(item, result);

      return result;
    },
    webhookUrl
  );

  return batchId;
}

// Use it
const batchId = await reviewWithAllFeatures([
  'Review this code...',
  'Check this design...',
  // ... 98 more
]);

console.log(`Batch ${batchId} submitted`);

// Monitor (this could run in background)
const checkBatch = async () => {
  const status = processor.getStatus(batchId);
  console.log(`Progress: ${status?.completedItems}/${status?.totalItems}`);

  if (status?.status === 'COMPLETE') {
    const results = processor.getResults(batchId);
    console.log(`Done! ${results?.length} reviews completed`);
  }
};
```

## 📝 API Reference

### CacheManager

```typescript
class CacheManager<T> {
  set(content: string, value: T): void
  get(content: string): T | null
  has(content: string): boolean
  clear(): void
  getStats(): {
    size: number
    maxSize: number
    ttl: number
    entries: Array<{
      hash: string
      createdAt: string
      expiresAt: string
      accessCount: number
      isExpired: boolean
    }>
  }
}
```

### RateLimiter

```typescript
class RateLimiter {
  waitIfNeeded(providerId: string): Promise<void>
  getState(providerId: string): RateLimiterState | null
  getAllStates(): Record<string, RateLimiterState>
  reset(providerId: string): void
}
```

### BatchProcessor

```typescript
class BatchProcessor<T, R> {
  submitBatch(
    items: T[],
    processor: (item: T, workerIndex: number) => Promise<R>,
    webhookUrl?: string
  ): Promise<string>
  
  getStatus(batchId: string): BatchJob<T, R> | null
  getResults(batchId: string): R[] | null
  listJobs(filter?: { status?: BatchStatus }): BatchJob[]
}
```
